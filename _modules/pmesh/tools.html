[docs]class Rotator(object):
    def __init__(self, comm):
        self.comm = comm
    def __enter__(self):
        self.comm.Barrier()
        for i in range(self.comm.rank):
            self.comm.Barrier()
    def __exit__(self, type, value, tb):
        for i in range(self.comm.rank, self.comm.size):
            self.comm.Barrier()
        self.comm.Barrier()


[docs]def FromRoot(comm):
    def decorator(func):
        def wrapped(*args, **kwargs):
            if comm.rank == 0:
                rt = func(*args, **kwargs)
            else:
                rt = None
            rt = comm.bcast(rt)
            return rt
        return wrapped
    return decorator

        t1 = 1.0 * MPI.Wtime()
        self.spent += t1 - self.t0
